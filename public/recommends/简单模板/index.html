判断质数

int is_prime(int num){
    for (int i = 2; i * i < num; i++)
    {
        if(num % i == 0) return -1;
    }
    return 1;
}

判断完全数

int is_perfect_number(int num){
    int sum = 0;
    for (int i = 1; i * i <= num; i++){
        if(num % i == 0){
            if(i < num) sum += i;
            if(num / i < num && i != num / i) sum += num / i;
        }
    }

    if(sum == num){
        return 1;
    }
    
    return -1;
}

数组翻转

template<typename T>
void reverse_arr(T *arr, int len){
    for (int i = 0; i < len / 2; i++) swap(arr[i], arr[len - i - 1]);
    
}

快速排序

void quick_sort(int* arr, int l, int r){ // l, r 为数组的左右边界，[l, r]
	if(l >= r) return ;
	int x = arr[l], i = l - 1, j = r + 1;
	while(i < j){
		do i++; while(arr[i] < x);
		do j--; while(arr[j] > x);
		if(i < j) swap(arr[i], arr[j]);
	}
	quick_sort(arr, l , j);
	quick_sort(arr, j + 1, r);
}

归并排序

int tmp[N]; // 临时数组

void merge_sort(int* arr, int l, int r){
	if(l >= r) return ;
    int mid = l + r >> 1;
    merge_sort(arr, l, mid);
    merge_sort(arr, mid + 1, r);
    int i = l, j = mid + 1, k = 0;
    while(i <= mid && j <= r)
        if(arr[i] <= arr[j]) tmp[k ++] = arr[i ++];
        else tmp[k ++] = arr[j ++];
    while(i <= mid) tmp[k ++] = arr[i ++];
    while(j <= r) tmp[k ++] = arr[j ++];

    for(int i = l, j = 0; i <= r;) arr[i ++] = tmp[k ++];
}

int quick_sort(int l, int r, int k){
    if(l >= r) return a[l];
    int i = l - 1, j = r + 1, x = a[l + r >> 1];
    while(i < j){
        do i++; while(a[i] < x);
        do j--; while(a[j] > x);
        if(i < j) swap(a[i], a[j]);
    }
    int sl = j - l + 1;
    if(k <= sl) return quick_sort(l, j, k);
    return quick_sort(j + 1, r, k - sl);
}
